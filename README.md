function pascal_triangle(MAXN)
    intialize a matrix dp[MAXN][MAXN] with 0<br>
    for i = 0 to MAXN<br>
        dp[i][0]=dp[0][i]=1<br>
    endfor<br>
    for i = 1 to MAXN<br>
        for j = 1 to MAXN<br>
            dp[i][j] = dp[i-1][j]+dp[i][j-1]<br>
        endfor<br>
    endfor
